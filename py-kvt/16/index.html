#3 Теорія: Шаблони Jinja2 у Flask

Типи тегів у Jinja2:
  • {{ "{{ ... }}" }} — виведення змінних та виразів
  • {{ "{% ... %}" }} — управляючі конструкції (if, for, block)
  • {{ "{# ... #}" }} — коментарі (не відображаються в HTML)
Приклад виведення змінних:
<!-- templates/hello.html -->
<h1>Привіт, {{ "{{ name }}" }}!</h1>
<p>Ваш вік: {{ "{{ age }}" }} років</p>
<p>Email: {{ "{{ user.email }}" }}</p>
<p>Перший елемент: {{ "{{ items[0] }}" }}</p>
Передача даних з Python:

Базовий синтаксис:
{{ "{% if user %}" }}
    <p>Привіт, {{ "{{ user.name }}" }}!</p>
{{ "{% else %}" }}
    <p>Привіт, гість!</p>
{{ "{% endif %}" }}
Використання elif:
{{ "{% if user.role == 'admin' %}" }}
    <span class="badge badge-danger">Адміністратор</span>
{{ "{% elif user.role == 'moderator' %}" }}
    <span class="badge badge-warning">Модератор</span>
{{ "{% else %}" }}
    <span class="badge badge-secondary">Користувач</span>
{{ "{% endif %}" }}
Оператори порівняння:
  • == — дорівнює
  • != — не дорівнює
  • <, >, <=, >= — порівняння
  • and, or, not — логічні оператори
  • in — перевірка входження
{{ "{% if age >= 18 and is_verified %}" }}
    <p>Доступ дозволено</p>
{{ "{% endif %}" }}

{{ "{% if 'admin' in user.roles %}" }}
    <a href="/admin">Панель адміністратора</a>
{{ "{% endif %}" }}

Базовий цикл:
<ul>
{{ "{% for item in items %}" }}
    <li>{{ "{{ item }}" }}</li>
{{ "{% endfor %}" }}
</ul>
Цикл з else (якщо список порожній):
<ul>
{{ "{% for product in products %}" }}
    <li>{{ "{{ product.name }}" }} - {{ "{{ product.price }}" }} грн</li>
{{ "{% else %}" }}
    <li>Товарів не знайдено</li>
{{ "{% endfor %}" }}
</ul>
Спеціальні змінні циклу (loop):
ЗміннаОпис
loop.indexНомер ітерації (починається з 1)
loop.index0Номер ітерації (починається з 0)
loop.firstTrue, якщо перша ітерація
loop.lastTrue, якщо остання ітерація
loop.lengthЗагальна кількість елементів
 
<table class="table">
{{ "{% for user in users %}" }}
    <tr class="{{ "{{ 'table-primary' if loop.first else '' }}" }}">
        <td>{{ "{{ loop.index }}" }}</td>
        <td>{{ "{{ user.name }}" }}</td>
    </tr>
{{ "{% endfor %}" }}
</table>
Ітерація по словнику:
{{ "{% for key, value in user.items() %}" }}
    <p><strong>{{ "{{ key }}" }}:</strong> {{ "{{ value }}" }}</p>
{{ "{% endfor %}" }}

Базовий шаблон (base.html):
<!DOCTYPE html>
<html lang="uk">
<head>
    <meta charset="UTF-8">
    <title>{{ "{% block title %}" }}Мій сайт{{ "{% endblock %}" }}</title>
    <link rel="stylesheet" href="{{ "{{ url_for('static', filename='css/style.css') }}" }}">
    {{ "{% block styles %}" }}{{ "{% endblock %}" }}
</head>
<body>
    <nav>
        <a href="/">Головна</a>
        <a href="/about">Про нас</a>
    </nav>
    
    <main class="container">
        {{ "{% block content %}" }}{{ "{% endblock %}" }}
    </main>
    
    <footer>
        <p>&copy; 2024 Мій сайт</p>
    </footer>
    
    {{ "{% block scripts %}" }}{{ "{% endblock %}" }}
</body>
</html>
Дочірній шаблон (index.html):

Використання super() для розширення блоку:
{{ "{% extends 'base.html' %}" }}

{{ "{% block styles %}" }}
    {{ "{{ super() }}" }}
    <link rel="stylesheet" href="{{ "{{ url_for('static', filename='css/custom.css') }}" }}">
{{ "{% endblock %}" }}

Популярні вбудовані фільтри:
ФільтрОписПриклад
upperВерхній регістр{{ "{{ name|upper }}" }}
lowerНижній регістр{{ "{{ name|lower }}" }}
capitalizeПерша літера велика{{ "{{ name|capitalize }}" }}
titleКожне слово з великої{{ "{{ text|title }}" }}
trimВидалити пробіли{{ "{{ text|trim }}" }}
lengthДовжина{{ "{{ items|length }}" }}
defaultЗначення за замовчуванням{{ "{{ name|default('Гість') }}" }}
safeНе екранувати HTML{{ "{{ html_content|safe }}" }}
joinОб'єднати список{{ "{{ items|join(', ') }}" }}
replaceЗамінити текст{{ "{{ text|replace('old', 'new') }}" }}
roundОкруглення{{ "{{ price|round(2) }}" }}
intПеретворити в число{{ "{{ value|int }}" }}
Комбінування фільтрів:
<p>{{ "{{ description|truncate(100)|capitalize }}" }}</p>
<p>{{ "{{ price|round(2)|string + ' грн' }}" }}</p>

Генерація URL за назвою функції:
# app.py
@app.route('/')
def index():
    return render_template('index.html')

@app.route('/user/<username>')
def profile(username):
    return render_template('profile.html', username=username)

@app.route('/post/<int:post_id>')
def post(post_id):
    return render_template('post.html', post_id=post_id)
<!-- Використання url_for в шаблоні -->
<a href="{{ "{{ url_for('index') }}" }}">Головна</a>
<a href="{{ "{{ url_for('profile', username='ivan') }}" }}">Профіль Івана</a>
<a href="{{ "{{ url_for('post', post_id=42) }}" }}">Стаття #42</a>

Створення макросу:
<!-- templates/macros/forms.html -->
{{ "{% macro input(name, label, type='text', value='', required=false) %}" }}
<div class="form-group">
    <label for="{{ "{{ name }}" }}">{{ "{{ label }}" }}</label>
    <input type="{{ "{{ type }}" }}" 
           class="form-control" 
           id="{{ "{{ name }}" }}" 
           name="{{ "{{ name }}" }}"
           value="{{ "{{ value }}" }}"
           {{ "{{ 'required' if required else '' }}" }}>
</div>
{{ "{% endmacro %}" }}

{{ "{% macro button(text, type='submit', class='btn-primary') %}" }}
<button type="{{ "{{ type }}" }}" class="btn {{ "{{ class }}" }}">{{ "{{ text }}" }}</button>
{{ "{% endmacro %}" }}
Використання макросів:
{{ "{% from 'macros/forms.html' import input, button %}" }}

<form method="POST">
    {{ "{{ input('username', \"Ім'я користувача\", required=true) }}" }}
    {{ "{{ input('email', 'Email', type='email', required=true) }}" }}
    {{ "{{ input('password', 'Пароль', type='password', required=true) }}" }}
    {{ "{{ button('Зареєструватися') }}" }}
</form>
Імпортування всіх макросів:

{{ "{{ forms.input('name', \"Ім'я\") }}" }}

Обробка форми у Flask:
from flask import Flask, render_template, request, redirect, url_for

app = Flask(__name__)

@app.route('/contact', methods=['GET', 'POST'])
def contact():
    if request.method == 'POST':
        name = request.form['name']
        email = request.form['email']
        message = request.form['message']
        
        # Обробка даних (збереження в БД, надсилання email тощо)
        print(f"Повідомлення від {name} ({email}): {message}")
        
        return redirect(url_for('thank_you'))
    
    return render_template('contact.html')

@app.route('/thank-you')
def thank_you():
    return render_template('thank_you.html')

Збереження введених даних при помилці:
@app.route('/register', methods=['GET', 'POST'])
def register():
    error = None
    if request.method == 'POST':
        username = request.form['username']
        if len(username) < 3:
            error = "Ім'я користувача занадто коротке"
            return render_template('register.html', error=error, username=username)
        # ... реєстрація
    return render_template('register.html', error=error)

Налаштування та використання:
from flask import Flask, flash, redirect, url_for, render_template

app = Flask(__name__)
app.secret_key = 'your-secret-key-here'  # Обов'язково для flash!

@app.route('/login', methods=['POST'])
def login():
    username = request.form['username']
    password = request.form['password']
    
    if username == 'admin' and password == 'secret':
        flash('Ви успішно увійшли!', 'success')
        return redirect(url_for('dashboard'))
    else:
        flash('Невірний логін або пароль', 'error')
        return redirect(url_for('login_page'))

@app.route('/logout')
def logout():
    flash('Ви вийшли з системи', 'info')
    return redirect(url_for('index'))
Відображення flash-повідомлень у шаблоні:
<!-- В base.html -->
{{ "{% with messages = get_flashed_messages(with_categories=true) %}" }}
    {{ "{% if messages %}" }}
        {{ "{% for category, message in messages %}" }}
            <div class="alert alert-{{ "{{ category }}" }} alert-dismissible fade show">
                {{ "{{ message }}" }}
                <button type="button" class="close" data-dismiss="alert">
                    <span>&times;</span>
                </button>
            </div>
        {{ "{% endfor %}" }}
    {{ "{% endif %}" }}
{{ "{% endwith %}" }}
Типи повідомлень (для Bootstrap):
  • success — зелене (успіх)
  • danger або error — червоне (помилка)
  • warning — жовте (попередження)
  • info — синє (інформація)
